Ink is written in TypeScript and exports comprehensive type definitions for all its APIs.

## Core Types

### RenderOptions

Configuration options for the `render()` function.

```typescript
type RenderOptions = {
  stdout?: NodeJS.WriteStream;
  stdin?: NodeJS.ReadStream;
  stderr?: NodeJS.WriteStream;
  debug?: boolean;
  exitOnCtrlC?: boolean;
  patchConsole?: boolean;
  onRender?: (metrics: RenderMetrics) => void;
  isScreenReaderEnabled?: boolean;
  maxFps?: number;
  incrementalRendering?: boolean;
  concurrent?: boolean;
  kittyKeyboard?: KittyKeyboardOptions;
  interactive?: boolean;
  alternateScreen?: boolean;
};
```

### Instance

Return type of the `render()` function.

```typescript
type Instance = {
  rerender: (node: ReactNode) => void;
  unmount: () => void;
  waitUntilExit: () => Promise<unknown>;
  waitUntilRenderFlush: () => Promise<void>;
  cleanup: () => void;
  clear: () => void;
};
```

### RenderToStringOptions

Configuration options for the `renderToString()` function.

```typescript
type RenderToStringOptions = {
  columns?: number; // default: 80
};
```

## Usage Examples

### Typed Component

```typescript
import React, {FC} from 'react';
import {Box, Text, BoxProps} from 'ink';

interface CardProps extends Partial<BoxProps> {
  title: string;
  children: React.ReactNode;
}

const Card: FC<CardProps> = ({title, children, ...boxProps}) => (
  <Box 
    flexDirection="column" 
    borderStyle="round" 
    padding={1}
    {...boxProps}
  >
    <Text bold>{title}</Text>
    <Box marginTop={1}>
      {children}
    </Box>
  </Box>
);
```

### Generic Static Component

```typescript
import React from 'react';
import {Static, Box, Text} from 'ink';

interface Task {
  id: number;
  name: string;
  completed: boolean;
}

const TaskList: React.FC<{tasks: Task[]}> = ({tasks}) => (
  <Static<Task> items={tasks}>
    {(task) => (
      <Box key={task.id}>
        <Text color={task.completed ? 'green' : 'yellow'}>
          {task.completed ? '✓' : '⌛'} {task.name}
        </Text>
      </Box>
    )}
  </Static>
);
```

### Typed Input Handler

```typescript
import {useInput, Key} from 'ink';

const MyComponent = () => {
  const handleInput = (input: string, key: Key) => {
    if (key.upArrow) {
      // Handle up arrow
    }
    if (key.return) {
      // Handle enter
    }
    if (input === 'q') {
      // Handle 'q' key
    }
  };

  useInput(handleInput);

  return <Text>Press any key</Text>;
};
```
