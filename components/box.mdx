title: Box

---

The `<Box>` component is the fundamental building block for layouts in Ink. It's equivalent to `<div style="display: flex">` in the browser and uses Flexbox for positioning.

## Basic Usage

```tsx
import {render, Box, Text} from 'ink';

const App = () => (
  <Box margin={2} padding={1} borderStyle="round">
    <Text>Hello from a Box!</Text>
  </Box>
);

render(<App />);
```

## Layout Properties

### Dimensions

<ParamField path="width" type="number | string">
  Width in spaces or as a percentage of parent width.
  
  ```tsx
  <Box width={10}>
    <Text>Fixed width</Text>
  </Box>
  
  <Box width="50%">
    <Text>Half width</Text>
  </Box>
  ```
</ParamField>

<ParamField path="height" type="number | string">
  Height in lines or as a percentage of parent height.
  
  ```tsx
  <Box height={5}>
    <Text>Fixed height</Text>
  </Box>
  
  <Box height="50%">
    <Text>Half height</Text>
  </Box>
  ```
</ParamField>

<ParamField path="minWidth" type="number">
  Minimum width of the element. Percentages not yet supported.
</ParamField>

<ParamField path="minHeight" type="number | string">
  Minimum height in lines or percentage of parent.
</ParamField>

<ParamField path="maxWidth" type="number">
  Maximum width of the element. Percentages not yet supported.
</ParamField>

<ParamField path="maxHeight" type="number | string">
  Maximum height in lines or percentage of parent.
</ParamField>

<ParamField path="aspectRatio" type="number">
  Aspect ratio (width/height). Use with at least one size constraint.
</ParamField>

## Flexbox Properties

<ParamField path="flexDirection" type="'row' | 'row-reverse' | 'column' | 'column-reverse'">
  Direction of flex items.
  
  ```tsx
  <Box flexDirection="column">
    <Text>First line</Text>
    <Text>Second line</Text>
  </Box>
  ```
</ParamField>

<ParamField path="flexGrow" type="number" default="0">
  How much the element should grow relative to siblings.
  
  ```tsx
  <Box>
    <Text>Label:</Text>
    <Box flexGrow={1}>
      <Text>Fills remaining space</Text>
    </Box>
  </Box>
  ```
</ParamField>

<ParamField path="flexShrink" type="number" default="1">
  How much the element should shrink relative to siblings.
</ParamField>

<ParamField path="flexBasis" type="number | string">
  Default size of element before remaining space is distributed.
</ParamField>

<ParamField path="flexWrap" type="'nowrap' | 'wrap' | 'wrap-reverse'">
  Whether flex items should wrap to new lines.
  
  ```tsx
  <Box width={10} flexWrap="wrap">
    <Text>Item 1 </Text>
    <Text>Item 2 </Text>
    <Text>Item 3</Text>
  </Box>
  ```
</ParamField>

<ParamField path="alignItems" type="'flex-start' | 'center' | 'flex-end' | 'stretch' | 'baseline'">
  Align items on cross axis.
  
  ```tsx
  <Box alignItems="center" height={5}>
    <Text>Centered vertically</Text>
  </Box>
  ```
</ParamField>

<ParamField path="alignSelf" type="'auto' | 'flex-start' | 'center' | 'flex-end' | 'stretch' | 'baseline'" default="'auto'">
  Override alignItems for individual element.
</ParamField>

<ParamField path="alignContent" type="'flex-start' | 'flex-end' | 'center' | 'stretch' | 'space-between' | 'space-around' | 'space-evenly'" default="'flex-start'">
  Alignment between flex lines when wrapping.
</ParamField>

<ParamField path="justifyContent" type="'flex-start' | 'center' | 'flex-end' | 'space-between' | 'space-around' | 'space-evenly'">
  Distribute items along main axis.
  
  ```tsx
  <Box justifyContent="space-between">
    <Text>Left</Text>
    <Text>Right</Text>
  </Box>
  ```
</ParamField>

### Gap

<ParamField path="gap" type="number" default="0">
  Gap between columns and rows. Shorthand for columnGap and rowGap.
  
  ```tsx
  <Box gap={1}>
    <Text>A</Text>
    <Text>B</Text>
    <Text>C</Text>
  </Box>
  ```
</ParamField>

<ParamField path="columnGap" type="number" default="0">
  Gap between columns.
</ParamField>

<ParamField path="rowGap" type="number" default="0">
  Gap between rows.
</ParamField>

### Borders

<ParamField path="borderStyle" type="'single' | 'double' | 'round' | 'bold' | 'singleDouble' | 'doubleSingle' | 'classic' | BoxStyle">
  Border style. Can be a preset name or a custom border object.
  
  ```tsx
  <Box borderStyle="round">
    <Text>Rounded border</Text>
  </Box>
  
  <Box borderStyle={{
    topLeft: '↘',
    top: '↓',
    topRight: '↙',
    left: '→',
    bottomLeft: '↗',
    bottom: '↑',
    bottomRight: '↖',
    right: '←'

    <Text>Custom border</Text>
  </Box>
  ```
</ParamField>

<ParamField path="borderColor" type="string">
  Border color. Shorthand for all border sides.
  

</ParamField>

<ParamField path="borderBottomColor" type="string">
  Bottom border color.

<ParamField path="borderLeftColor" type="string">
  Left border color.
</ParamField>

<ParamField path="borderDimColor" type="boolean" default="false">
  Dim all border colors.
</ParamField>

<ParamField path="borderTop" type="boolean" default="true">
  Show top border.
</ParamField>

<ParamField path="borderRight" type="boolean" default="true">

</ParamField>

<ParamField path="borderBottom" type="boolean" default="true">

<ParamField path="borderLeft" type="boolean" default="true">

### Background

<ParamField path="backgroundColor" type="string">
  Background color for the box. Fills the entire area and is inherited by child `<Text>` components.
  
  ```tsx
  <Box backgroundColor="blue" padding={1}>
    <Text>Text on blue background</Text>
  </Box>
  ```
</ParamField>

## Examples

### Border Styles

From the Ink examples:

```tsx
import {Box, Text} from 'ink';

const Borders = () => (
  <Box flexDirection="column" padding={2}>
    <Box>
      <Box borderStyle="single" marginRight={2}>
        <Text>single</Text>
      </Box>
      <Box borderStyle="double" marginRight={2}>
        <Text>double</Text>
      </Box>
      <Box borderStyle="round" marginRight={2}>
        <Text>round</Text>
      </Box>
      <Box borderStyle="bold">
        <Text>bold</Text>
      </Box>
    </Box>
  </Box>
);
```

### Flexbox Layout

```tsx
import {Box, Text, Spacer} from 'ink';

const Layout = () => (
  <Box flexDirection="column" padding={1}>
    {/* Header with space-between */}
    <Box justifyContent="space-between">
      <Text bold>My App</Text>
      <Text dimColor>v1.0.0</Text>
    </Box>
    
    {/* Content with spacer */}
    <Box marginTop={1}>
      <Text>Left aligned</Text>
      <Spacer />
      <Text>Right aligned</Text>
    </Box>
    
    {/* Centered content */}
    <Box justifyContent="center" marginTop={1}>
      <Text>Centered text</Text>
    </Box>
  </Box>
);
```

### Background Colors

From the Ink examples:

```tsx
import {Box, Text} from 'ink';

const Backgrounds = () => (
  <Box flexDirection="column" gap={1}>
    <Box backgroundColor="red" width={20} height={3}>
      <Text>Red background</Text>
    </Box>
    
    <Box backgroundColor="#FF8800" width={20} height={3}>
      <Text>Orange background</Text>
    </Box>
    
    <Box backgroundColor="rgb(0, 255, 0)" width={20} height={3}>
      <Text>Green background</Text>
    </Box>
    
    {/* Background inheritance */}
    <Box backgroundColor="cyan">
      <Text>Inherited </Text>
      <Text backgroundColor="yellow">Override </Text>
      <Text>Back to inherited</Text>
    </Box>
  </Box>
);
```
