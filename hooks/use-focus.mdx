---
title: 'useFocus'
description: 'React hook for managing component focus state in Ink'
---

`useFocus` is a React hook that returns focus state and focus controls for the current component. A component that uses `useFocus` becomes "focusable" to Ink, so when the user presses <kbd>Tab</kbd>, Ink will switch focus to this component.

If there are multiple components that execute the `useFocus` hook, focus will be given to them in the order in which these components are rendered.

## Usage

```tsx
import {useFocus, Text} from 'ink';

const Example = () => {
  const {isFocused} = useFocus();

  return <Text>{isFocused ? 'I am focused' : 'I am not focused'}</Text>;
};
```

## Parameters

<ParamField path="options" type="object" optional>
  Configuration options for focus behavior.

  <Expandable title="properties">
    <ParamField path="isActive" type="boolean" default="true">
      Enable or disable this component's focus, while still maintaining its position in the list of focusable components. This is useful for inputs that are temporarily disabled.
    </ParamField>

    <ParamField path="autoFocus" type="boolean" default="false">
      Auto-focus this component if there's no active (focused) component right now.
    </ParamField>

    <ParamField path="id" type="string" optional>
      Assign an ID to this component, so it can be programmatically focused with `focus(id)`. This is useful for large interfaces with many focusable elements to avoid having to cycle through all of them.
    </ParamField>
  </Expandable>
</ParamField>

## Return Value

Returns an object with the following properties:

<ResponseField name="isFocused" type="boolean">
  Determines whether this component is currently focused.
</ResponseField>

<ResponseField name="focus" type="(id: string) => void">
  Allows focusing a specific element with the provided `id`.
</ResponseField>

## Examples

### Basic Focus

```tsx
import React from 'react';
import {Box, Text, render, useFocus} from 'ink';

function Focus() {
  return (
    <Box flexDirection="column" padding={1}>
      <Box marginBottom={1}>
        <Text>
          Press Tab to focus next element, Shift+Tab to focus previous element,
          Esc to reset focus.
        </Text>
      </Box>
      <Item label="First" />
      <Item label="Second" />
      <Item label="Third" />
    </Box>
  );
}

function Item({label}) {
  const {isFocused} = useFocus();
  return (
    <Text>
      {label} {isFocused ? <Text color="green">(focused)</Text> : null}
    </Text>
  );
}

render(<Focus />);
```

### Programmatic Focus with IDs

```tsx
import React from 'react';
import {
  render,
  Box,
  Text,
  useFocus,
  useInput,
  useFocusManager,
} from 'ink';

function Focus() {
  const {focus} = useFocusManager();

  useInput(input => {
    if (input === '1') {
      focus('1');
    }

    if (input === '2') {
      focus('2');
    }

    if (input === '3') {
      focus('3');
    }
  });

  return (
    <Box flexDirection="column" padding={1}>
      <Box marginBottom={1}>
        <Text>
          Press Tab to focus next element, Shift+Tab to focus previous element,
          Esc to reset focus.
        </Text>
      </Box>
      <Item id="1" label="Press 1 to focus" />
      <Item id="2" label="Press 2 to focus" />
      <Item id="3" label="Press 3 to focus" />
    </Box>
  );
}

function Item({label, id}) {
  const {isFocused} = useFocus({id});

  return (
    <Text>
      {label} {isFocused ? <Text color="green">(focused)</Text> : null}
    </Text>
  );
}

render(<Focus />);
```

### Disabled Focus

```tsx
import {useState} from 'react';
import {Box, Text, useFocus, useInput} from 'ink';

const Input = ({disabled}) => {
  const {isFocused} = useFocus({isActive: !disabled});

  return (
    <Text color={disabled ? 'gray' : isFocused ? 'green' : 'white'}>
      {disabled ? '[Disabled]' : '[Active]'} Input
    </Text>
  );
};
```
